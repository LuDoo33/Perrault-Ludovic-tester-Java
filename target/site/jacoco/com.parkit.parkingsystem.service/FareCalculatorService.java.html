        int inMinute = ticket.getInTime().getMinutes();
        int outMinute = ticket.getOutTime().getMinutes();

        double inTime = (((double)inDay * 24) + inHour + ((double)inMinute / 60));
        double outTime = (((double)outDay * 24) + outHour + ((double)outMinute / 60));
